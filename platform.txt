### Made possible through sleepless nights
## Metadata
name=Nuvoton N76E003
version=0.1.0

## Tool paths
bin.path={runtime.tools.bundle.path}/bin
device.path={runtime.tools.bundle.path}/device

## Compiler
compiler.sdcc.cmd=sdcc
compiler.sdcc.cmd.windows=sdcc.exe
compiler.ar.cmd=sdar
compiler.ar.cmd.windows=sdar.exe
compiler.objcopy.cmd=sdobjcopy
compiler.objcopy.cmd.windows=sdobjcopy.exe
tools.nvtispflash.cmd=nvtispflash
tools.nvtispflash.cmd.windows=nvtispflash.exe

## Global compiler flags
## Patch: SDCC accepts .o as .rel and .a as .lib
compiler.sdcc.flags=-mmcs51 --model-{build.model} --iram-size 256 --xram-size 768 --code-size 18432 -DF_CPU={build.f_cpu} -I"{device.path}/include/mcs51" -I"{device.path}/include" --disable-warning 283 --disable-warning 126

## Linker flags (reused in combine)
compiler.sdcc.ld.flags=-mmcs51 --model-{build.model} --iram-size 256 --xram-size 768 --code-size 18432 -L"{device.path}/lib/{build.model}" --out-fmt-ihx

## Compile C
recipe.c.o.pattern="{bin.path}/{compiler.sdcc.cmd}" {compiler.sdcc.flags} {includes} -c "{source_file}" -o "{object_file}"

## Compile sketch (.ino → .cpp, forced as C)
recipe.cpp.o.pattern="{bin.path}/{compiler.sdcc.cmd}" {compiler.sdcc.flags} {includes} -x c -c "{source_file}" -o "{object_file}"

## Preprocess (for library detection)
recipe.preproc.includes.pattern="{bin.path}/{compiler.sdcc.cmd}" {compiler.sdcc.flags} {includes} -x c -E "{source_file}"
recipe.preproc.macros.pattern="{bin.path}/{compiler.sdcc.cmd}" {compiler.sdcc.flags} {includes} -x c -E "{source_file}"

## Archive core into core.a
recipe.ar.pattern="{bin.path}/{compiler.ar.cmd}" -rc "{archive_file_path}" "{object_file}"

## Link
## Explicitly pass main.c.o so SDCC sees main() and generates startup code
recipe.c.combine.pattern="{bin.path}/{compiler.sdcc.cmd}" {compiler.sdcc.ld.flags} "{build.path}/core/main.c.o" {object_files} "{archive_file_path}" -o "{build.path}/{build.project_name}.ihx"

## Convert ihx → bin
compiler.objcopy.bin.flags=-I ihex -O binary
recipe.objcopy.bin.pattern="{bin.path}/{compiler.objcopy.cmd}" {compiler.objcopy.bin.flags} "{build.path}/{build.project_name}.ihx" "{build.path}/{build.project_name}.bin"

## Size (optional, reads from SDCC .mem file)
recipe.size.pattern=
recipe.size.regex=
recipe.size.regex.data=
recipe.size.regex.eeprom=

## Upload
tools.nvtispflash.upload.params.verbose=
tools.nvtispflash.upload.params.quiet=
tools.nvtispflash.upload.pattern="{bin.path}/{tools.nvtispflash.cmd}" --serial-device "{serial.port}" --aprom-file "{build.path}/{build.project_name}.bin"
