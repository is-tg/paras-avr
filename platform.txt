## Metadata
name=Nuvoton N76E003
version=0.1.0

bin.path={runtime.tools.bundle.path}/bin
share.path={runtime.tools.bundle.path}/share/sdcc
lib.path={share.path}/lib/{build.model}

## Compiler
compiler.system.includes=-I"{share.path}/include/{build.arch}" -I"{share.path}/include"
compiler.sdcc.cmd=sdcc
compiler.sdcc.cmd.windows=sdcc.exe

## Global compiler flags
compiler.sdcc.flags=-m{build.arch} --model-{build.model} --iram-size 256 --xram-size 768 --code-size 18432 -DF_CPU={build.f_cpu} {compiler.system.includes}

## Compile C to object/rel files
recipe.c.o.pattern="{bin.path}/{compiler.sdcc.cmd}" {compiler.sdcc.flags} {includes} -c "{source_file}" -o "{object_file}"

## Sketch .ino becomes .cpp & force SDCC to treat as C
recipe.cpp.o.pattern="{bin.path}/{compiler.sdcc.cmd}" {compiler.sdcc.flags} {includes} -x c -c "{source_file}" -o "{object_file}"

## Force Arduino to use SDCC for dependency generation (Added -x c here too)
recipe.preproc.includes.pattern="{bin.path}/{compiler.sdcc.cmd}" -E -x c {compiler.sdcc.flags} {includes} "{source_file}"
recipe.preproc.macros.pattern="{bin.path}/{compiler.sdcc.cmd}" -E -x c {compiler.sdcc.flags} {includes} "{source_file}"

## Annoyance
compiler.ar.cmd=echo
recipe.ar.pattern="{compiler.ar.cmd}" "Skipping archive for {object_file}"

## Link using collected .rel object files
recipe.c.combine.pattern="{bin.path}/{compiler.sdcc.cmd}" -m{build.arch} --model-{build.model} --iram-size 256 --xram-size 768 --code-size 18432 -L{lib.path} --out-fmt-ihx "{build.path}/core/main.c.o" "{build.path}/core/wiring.c.o" "{build.path}/core/wiring_digital.c.o" {object_files} -o "{build.path}/{build.project_name}.ihx"

## Convert ihx -> bin
compiler.objcopy.cmd=sdobjcopy
compiler.objcopy.cmd.windows=sdobjcopy.exe
compiler.objcopy.bin.flags=-I ihex -O binary 
recipe.objcopy.bin.pattern="{bin.path}/{compiler.objcopy.cmd}" {compiler.objcopy.bin.flags} "{build.path}/{build.project_name}.ihx" "{build.path}/{build.project_name}.bin"

## Flash
tools.nvtispflash.cmd=nvtispflash
tools.nvtispflash.cmd.windows=nvtispflash.exe
tools.nvtispflash.upload.pattern="{bin.path}/{tools.nvtispflash.cmd}" --serial-device "{serial.port}" --aprom-file "{build.path}/{build.project_name}.bin"
