## Metadata
name=Paras 8051 Boards
version=0.0.1

## Define base SDCC path to cure ugliness
sdcc.path={runtime.tools.bundle.path}

## Compiler
compiler.sdcc.cmd=sdcc
compiler.sdcc.cmd.windows=sdcc.exe

## Define paths
compiler.path={sdcc.path}/bin
compiler.lib.path={sdcc.path}/share/sdcc/lib/{build.model}
compiler.system.includes=-I"{sdcc.path}/share/sdcc/include/{build.mcu}" -I"{sdcc.path}/share/sdcc/include"

## Suppress annoying SDCC warnings (283: no prototype, 126: unreachable code)
compiler.warning.flags=--disable-warning 283 --disable-warning 126

## Global compiler flags
compiler.sdcc.flags=-m{build.mcu} --model-{build.model} --iram-size 256 --xram-size 768 --code-size 18432 -DF_CPU={build.f_cpu} {compiler.system.includes} {compiler.warning.flags}

## Compile C to object/rel files
recipe.c.o.pattern="{compiler.path}/{compiler.sdcc.cmd}" {compiler.sdcc.flags} {includes} -c "{source_file}" -o "{object_file}"

## Sketch .ino becomes .cpp & force SDCC to treat as C
recipe.cpp.o.pattern="{compiler.path}/{compiler.sdcc.cmd}" {compiler.sdcc.flags} {includes} -x c -c "{source_file}" -o "{object_file}"

## Force Arduino to use SDCC for dependency generation (Added -x c here too)
recipe.preproc.includes.pattern="{compiler.path}/{compiler.sdcc.cmd}" -E -x c {compiler.sdcc.flags} {includes} "{source_file}"
recipe.preproc.macros.pattern="{compiler.path}/{compiler.sdcc.cmd}" -E -x c {compiler.sdcc.flags} {includes} "{source_file}"

## Annoyance
compiler.ar.cmd=echo
recipe.ar.pattern="{compiler.ar.cmd}" "Skipping archive for {object_file}"

## Link using collected .rel object files
recipe.c.combine.pattern="{compiler.path}/{compiler.sdcc.cmd}" -m{build.mcu} --model-{build.model} --iram-size 256 --xram-size 768 --code-size 18432 -L{compiler.lib.path} --out-fmt-ihx "{build.path}/core/main.c.o" "{build.path}/core/wiring.c.o" "{build.path}/core/wiring_digital.c.o" {object_files} -o "{build.path}/{build.project_name}.ihx"

## Convert ihx -> bin
compiler.objcopy.cmd=sdobjcopy
compiler.objcopy.cmd.windows=sdobjcopy.exe
compiler.objcopy.bin.flags=-I ihex -O binary 
recipe.objcopy.bin.pattern="{compiler.path}/{compiler.objcopy.cmd}" {compiler.objcopy.bin.flags} "{build.path}/{build.project_name}.ihx" "{build.path}/{build.project_name}.bin"

## Flash
tools.nvtispflash.cmd=nvtispflash
tools.nvtispflash.cmd.windows=nvtispflash.exe
tools.nvtispflash.upload.pattern="{compiler.path}/{tools.nvtispflash.cmd}" --serial-device "{serial.port}" --aprom-file "{build.path}/{build.project_name}.bin"
